📌 Review Sesi 9 – Extension Methods

  • Extension methods: menambahkan fungsi ke tipe yang sudah ada tanpa mengubah kode asli.
  • Sintaks: extension Nama on Type { ... }
  • Contoh: 'hello'.capitalize()

Contoh cepat:

extension IntExt on int {
    bool get isEven => this % 2 == 0;
}

void main() {
    print(4.isEven); // true
    print(5.isEven); // false
}

🧠 Pengantar Factory Constructors

  • Apa itu? Constructor khusus yang tidak selalu membuat instance baru.
  • Keyword: factory
  • Beda dengan generative constructor:
    • Generative: Selalu membuat instance baru.
    • Factory: Bisa mengembalikan instance yang sudah ada, atau instance dari subclass.
💡 Inti: Factory constructor memberi fleksibilitas dalam pembuatan objek — tidak terikat pada pembuatan instance baru setiap kali dipanggil.

📐 Sintaks Factory Constructor

class MyClass {
    // Generative constructor (biasa)
    MyClass();

    // Factory constructor
    factory MyClass.create() {
        // Bisa return instance yang sudah ada
        // Atau instance dari subclass
        // Atau instance baru
        return MyClass();
    }
}

void main() {
    // Perbedaan penggunaan
    var obj1 = MyClass();      // Generative
    var obj2 = MyClass.create(); // Factory
}

📌 Factory constructor harus mengembalikan instance dari tipe yang sama atau subclass-nya.

🔒 Kasus 1 – Singleton Pattern

Singleton: Hanya satu instance sepanjang aplikasi.

class DatabaseConnection {
    // Private constructor
    DatabaseConnection._internal() {
        print('Creating database connection...');
    }

    // Static instance
    static final DatabaseConnection _instance = DatabaseConnection._internal();

    // Factory constructor
    factory DatabaseConnection() {
        return _instance;
    }

    void connect() {
        print('Connected to database');
    }
}

void main() {
    // Penggunaan
    var db1 = DatabaseConnection();
    var db2 = DatabaseConnection();
    print(identical(db1, db2)); // true - instance yang sama
    db1.connect();
    db2.connect();
}
✅ Dengan factory constructor, kita memastikan hanya satu instance yang dibuat dan digunakan di seluruh aplikasi.

🗄️ Kasus 2 – Factory dengan Caching (ConnectionPool)

Cache koneksi berdasarkan parameter host dan port.

class ConnectionPool {
    final String host;
    final int port;

    // Cache berdasarkan parameter
    static final Map<String, ConnectionPool> _cache = {};

    // Private constructor
    ConnectionPool._(this.host, this.port);

    // Factory dengan caching
    factory ConnectionPool(String host, int port) {
        var key = '$host:$port';
        // Return dari cache jika ada
        if (_cache.containsKey(key)) {
            print('Returning cached connection for $key');
            return _cache[key]!;
        }
        // Buat baru dan cache
        print('Creating new connection for $key');
        var connection = ConnectionPool._(host, port);
        _cache[key] = connection;
        return connection;
    }

    void query(String sql) {
        print('Executing: "$sql" on $host:$port');
    }
}

void main() {
    print('-- Creating conn1 (localhost:5432) --');
    var conn1 = ConnectionPool('localhost', 5432);
    conn1.query('SELECT * FROM users;');

    print('\n-- Creating conn2 (localhost:5432 - should be cached) --');
    var conn2 = ConnectionPool('localhost', 5432);
    conn2.query('INSERT INTO products (name) VALUES (\'Laptop\');');

    print('\n-- Creating conn3 (db.example.com:3306 - new connection) --');
    var conn3 = ConnectionPool('db.example.com', 3306);
    conn3.query('UPDATE orders SET status=\'shipped\' WHERE id=101');

    print('\n-- Are conn1 and conn2 the same instance? ${identical(conn1, conn2)}');
    print('-- Are conn1 and conn3 the same instance? ${identical(conn1, conn3)}');
}
🗂️ Factory constructor memeriksa cache terlebih dahulu. Jika sudah ada, kembalikan instance yang sama — efisien untuk resource-heavy objects.

🔀 Kasus 3 – Mengembalikan Subclass (Shape)

Factory constructor bisa mengembalikan subclass berdasarkan parameter input.

abstract class Shape {
    void draw();

    // Factory constructor
    factory Shape(String type) {
        switch (type) {
            case 'circle':
                return Circle();
            case 'square':
                return Square();
            case 'triangle':
                return Triangle();
            case 'rectangle':
                return Rectangle();
            default:
                throw ArgumentError('Unknown shape: $type');
        }
    }
}

class Circle implements Shape {
    @override
    void draw() => print('Drawing Circle');
}

class Square implements Shape {
    @override
    void draw() => print('Drawing Square');
}

class Triangle implements Shape {
    @override
    void draw() => print('Drawing Triangle');
}

class Rectangle implements Shape {
    @override
    void draw() => print('Drawing Rectangle');
}

void main() {
    var circle = Shape('circle');
    var square = Shape('square');
    var triangle = Shape('triangle');
    var rectangle = Shape('rectangle');

    circle.draw();   // Drawing Circle
    square.draw();   // Drawing Square
    triangle.draw(); // Drawing Triangle
    rectangle.draw(); // Drawing Rectangle

    // var unknown = Shape('pentagon'); // throws ArgumentError
}
🔀 Factory constructor bertindak sebagai factory method yang memilih subclass yang tepat berdasarkan input, menjaga kode klien tetap sederhana.

⚙️ Praktik 1 – Factory untuk Configuration

Membuat factory untuk objek Config dengan berbagai source (file, environment, default).

class Config {
    final String apiUrl;
    final int timeout;
    final bool debug;

    // Private constructor
    Config._({required this.apiUrl, required this.timeout, required this.debug});

    // Factory dari file JSON (simulasi)
    factory Config.fromJson(Map<String, dynamic> json) {
        return Config._(
            apiUrl: json['apiUrl'] ?? 'https://default.api',
            timeout: json['timeout'] ?? 30,
            debug: json['debug'] ?? false,
        );
    }

    // Factory dari environment (simulasi)
    factory Config.fromEnv() {
        // Dalam realita, membaca dari Platform.environment
        return Config._(
            apiUrl: 'https://env.api',
            timeout: 60,
            debug: true,
        );
    }

    // Factory default
    factory Config.defaultConfig() {
        return Config._(
            apiUrl: 'https://default.api',
            timeout: 30,
            debug: false,
        );
    }

    @override
    String toString() => 'Config(apiUrl: $apiUrl, timeout: $timeout, debug: $debug)';
}

void main() {
    var config1 = Config.fromJson({'apiUrl': 'https://prod.api', 'timeout': 45});
    var config2 = Config.fromEnv();
    var config3 = Config.defaultConfig();

    print(config1);
    print(config2);
    print(config3);
}
💡 Factory constructor memungkinkan pembuatan objek dari berbagai sumber dengan cara yang konsisten dan terstruktur.

📦 Response Class – Success / Error

Factory constructor untuk membuat Response<T> yang sukses atau error.

abstract class Response<T> {
    final T? data;
    final String? error;

    Response({this.data, this.error});

    // Factory untuk success
    factory Response.success(T data) = SuccessResponse<T>;

    // Factory untuk error
    factory Response.error(String error) = ErrorResponse<T>;
}

class SuccessResponse<T> extends Response<T> {
    SuccessResponse(T data) : super(data: data);
}

class ErrorResponse<T> extends Response<T> {
    ErrorResponse(String error) : super(error: error);
}

void main() {
    var success = Response<String>.success('Data berhasil dimuat');
    var error = Response<int>.error('Gagal memuat data');

    print('Success: data=${success.data}, error=${success.error}');
    print('Error: data=${error.data}, error=${error.error}');
}
📦 Pola ini sangat umum di API handling — memisahkan respons sukses dan error dengan tipe yang aman.

⚖️ Factory Constructor vs Static Method

Perbedaan utama: Factory constructor dapat digunakan di new expression dan mendukung polymorphism.

class Example {
    Example._() {}

    // Factory constructor
    factory Example.create() => Example._();

    // Static method
    static Example createStatic() => Example._();
}

void main() {
    // Penggunaan sama secara sintaks
    var e1 = Example.create();       // Factory constructor
    var e2 = Example.createStatic(); // Static method

    print('e1: $e1');
    print('e2: $e2');
    print('Apakah e1 dan e2 berbeda? ${e1 != e2}');
}
AspekFactory ConstructorStatic Method
Digunakan dengan new✅ Ya❌ Tidak
Dapat mengembalikan subclass✅ Ya✅ Ya
Mendukung inheritance✅ Ya❌ Tidak
Membuat instance baru⚡ Opsional⚡ Opsional
💡 Gunakan factory constructor jika ingin objek dibuat dengan syntax seperti constructor dan mendukung polimorfisme. Gunakan static method untuk utility sederhana.

🎨 Color Class dengan Cache

Menggunakan factory constructor untuk meng-cache objek Color yang immutable.

class Color {
    final int red;
    final int green;
    final int blue;

    // Cache immutable objects
    static final Map<String, Color> _cache = {};

    // Private constructor
    Color._(this.red, this.green, this.blue);

    // Factory dengan cache
    factory Color.rgb(int red, int green, int blue) {
        // Validasi
        if (red < 0 || red > 255 ||
            green < 0 || green > 255 ||
            blue < 0 || blue > 255) {
            throw ArgumentError('RGB values must be 0-255');
        }
        var key = '$red,$green,$blue';
        if (!_cache.containsKey(key)) {
            _cache[key] = Color._(red, green, blue);
        }
        return _cache[key]!;
    }

    // Named factories untuk warna umum
    factory Color.red() => Color.rgb(255, 0, 0);
    factory Color.green() => Color.rgb(0, 255, 0);
    factory Color.blue() => Color.rgb(0, 0, 255);
    factory Color.white() => Color.rgb(255, 255, 255);
    factory Color.black() => Color.rgb(0, 0, 0);

    @override
    String toString() => 'Color($red, $green, $blue)';
}

void main() {
    var red1 = Color.red();
    var red2 = Color.rgb(255, 0, 0);
    print(identical(red1, red2)); // true - instance sama
    print(red1);
    print(red2);
}
🎯 Dengan cache, warna yang sama tidak dibuat berulang kali — menghemat memori untuk objek yang immutable dan sering digunakan.

📝 Ringkasan & Best Practices

Ringkasan

  • Factory constructor: factory Class.name()
  • Bisa mengembalikan: instance yang ada, subclass, atau instance baru.
  • Use cases: Singleton, caching, object pooling, subclass selection.

Best Practices

  • Gunakan factory untuk:
    • Object pooling / caching
    • Singleton pattern
    • Complex object creation
    • Subclass selection
  • Gunakan generative constructor untuk:
    • Simple object creation
    • Initialization lists
    • Required parameters
  • Validation: Lakukan validasi di factory sebelum membuat objek.
⚡ Ingat: Factory constructor tidak memiliki akses ke this sebelum objek dibuat — ia hanya bertugas mengembalikan instance.

Buat ConnectionFactory dengan pooling koneksi database.

class ConnectionFactory {
    final String host;
    final int port;
    final int maxPoolSize;

    static final Map<String, ConnectionFactory> _pool = {};

    ConnectionFactory._(this.host, this.port, this.maxPoolSize);

Buat NotificationFactory yang memilih subclass berdasarkan platform.

abstract class Notification {
    void send(String message);
}

class EmailNotification implements Notification {
    @override
    void send(String message) {
        print('Sending email: $message');
    }
}

class SMSNotification implements Notification {
    @override
    void send(String message) {
        print('Sending SMS: $message');
    }

class PushNotification implements Notification {
    @override
    void send(String message) {
        print('Sending push notification: $message');
    }

Buat ShapeFactory dengan cache untuk shapes yang sama.

abstract class Shape {
    void draw();
    String get type;

class Circle implements Shape {
    final double radius;

🔥 Challenge AnimalFactory

Buat AnimalFactory dengan:

  • Factory method untuk membuat berbagai jenis hewan
  • Cache untuk hewan dengan atribut sama
  • Validation untuk parameter
  • Subclass selection berdasarkan input
✅ Jawaban Challenge: 
abstract class Animal {
    String get name;
    String speak();
}

class Dog implements Animal {
    final String _name;
    Dog(this._name) {
        if (_name.isEmpty) throw ArgumentError('Name cannot be empty');
    }
    @override
    String get name => _name;
    @override
    String speak() => 'Woof!';
}

class Cat implements Animal {
    final String _name;
    Cat(this._name) {
        if (_name.isEmpty) throw ArgumentError('Name cannot be empty');
    }
    @override
    String get name => _name;
    @override
    String speak() => 'Meow!';
}

class Cow implements Animal {
    final String _name;
    Cow(this._name) {
        if (_name.isEmpty) throw ArgumentError('Name cannot be empty');
    }
    @override
    String get name => _name;
    @override
    String speak() => 'Moo!';
}

class AnimalFactory {
    static final Map<String, Animal> _cache = {};

    static Animal create(String type, String name) {
        if (name.isEmpty) {
            throw ArgumentError('Animal name cannot be empty');
        }

        var key = '$type:$name';

        if (_cache.containsKey(key)) {
            return _cache[key]!;
        }

        Animal animal;
        switch (type.toLowerCase()) {
            case 'dog':
                animal = Dog(name);
                break;
            case 'cat':
                animal = Cat(name);
                break;
            case 'cow':
                animal = Cow(name);
                break;
            default:
                throw ArgumentError('Unknown animal type: $type');
        }

        _cache[key] = animal;
        return animal;
    }
}

void main() {
    // Test cache
    var dog1 = AnimalFactory.create('dog', 'Buddy');
    var dog2 = AnimalFactory.create('dog', 'Buddy');
    var dog3 = AnimalFactory.create('dog', 'Rex');

    print(identical(dog1, dog2)); // true
    print(identical(dog1, dog3)); // false

    print('${dog1.name} says ${dog1.speak()}');
    print('${dog3.name} says ${dog3.speak()}');

    // Test cat
    var cat = AnimalFactory.create('cat', 'Whiskers');
    print('${cat.name} says ${cat.speak()}');

    // Test cow
    var cow = AnimalFactory.create('cow', 'Bessie');
    print('${cow.name} says ${cow.speak()}');

    // Test validation
    try {
        AnimalFactory.create('dog', '');
    } catch (e) {
        print('Validation works: $e');
    }

    // Test unknown type
    try {
        AnimalFactory.create('bird', 'Tweety');
    } catch (e) {
        print('Unknown type handled: $e');
    }
}
