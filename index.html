🧠 Pengantar Factory Constructors

  • Apa itu? Constructor khusus yang tidak selalu membuat instance baru.
  • Keyword: factory
  • Beda dengan generative constructor:
    • Generative: Selalu membuat instance baru.
    • Factory: Bisa mengembalikan instance yang sudah ada, atau instance dari subclass.
💡 Inti: Factory constructor memberi fleksibilitas dalam pembuatan objek — tidak terikat pada pembuatan instance baru setiap kali dipanggil.

📐 Sintaks Factory Constructor

class MyClass {
  // Generative constructor (biasa)
  MyClass();

  // Factory constructor
  factory MyClass.create() {
    // Bisa mengembalikan instance yang sudah ada,
    // instance subclass, atau instance baru
    return MyClass();
  }
}

void main() {
  // Perbedaan penggunaan
  var obj1 = MyClass();         // Generative constructor
  var obj2 = MyClass.create();  // Factory constructor

  print(obj1);
  print(obj2);
}

📌 Factory constructor harus mengembalikan instance dari tipe yang sama atau subclass-nya.

🔒 Kasus 1 – Singleton Pattern

Singleton: Hanya satu instance sepanjang aplikasi.

class DatabaseConnection {
    // Private constructor
    DatabaseConnection._internal() {
        print('Creating database connection...');
    }

    // Static instance
    static final DatabaseConnection _instance = DatabaseConnection._internal();

    // Factory constructor
    factory DatabaseConnection() {
        return _instance;
    }

    void connect() {
        print('Connected to database');
    }
}

void main() {
    // Penggunaan
    var db1 = DatabaseConnection();
    var db2 = DatabaseConnection();
    print(identical(db1, db2)); // true - instance yang sama
    db1.connect();
    db2.connect();
}
✅ Dengan factory constructor, kita memastikan hanya satu instance yang dibuat dan digunakan di seluruh aplikasi.

🗄️ Kasus 2 – Factory dengan Caching (ConnectionPool)

Cache koneksi berdasarkan parameter host dan port.

class ConnectionPool {
  final String host;
  final int port;

  // Cache berdasarkan parameter
  static final Map _cache = {};

  // Private constructor
  ConnectionPool._(this.host, this.port);

  // Factory dengan caching
  factory ConnectionPool(String host, int port) {
    var key = '$host:$port';

    // Return dari cache jika ada
    if (_cache.containsKey(key)) {
      print('Returning cached connection for $key');
      return _cache[key]!;
    }

    // Buat baru dan cache
    print('Creating new connection for $key');
    var connection = ConnectionPool._(host, port);
    _cache[key] = connection;

    return connection;
  }

  void query(String sql) {
    print('Executing: "$sql" on $host:$port');
  }
}

void main() {
  print('-- Creating conn1 (localhost:5432) --');
  var conn1 = ConnectionPool('localhost', 5432);
  conn1.query('SELECT * FROM users;');

  print('\n-- Creating conn2 (localhost:5432 - should be cached) --');
  var conn2 = ConnectionPool('localhost', 5432);
  conn2.query("INSERT INTO products (name) VALUES ('Laptop');");

  print('\n-- Creating conn3 (db.example.com:3306 - new connection) --');
  var conn3 = ConnectionPool('db.example.com', 3306);
  conn3.query("UPDATE orders SET status='shipped' WHERE id=101");

  print('\n-- Are conn1 and conn2 the same instance? ${identical(conn1, conn2)}');
  print('-- Are conn1 and conn3 the same instance? ${identical(conn1, conn3)}');
}
🗂️ Factory constructor memeriksa cache terlebih dahulu. Jika sudah ada, kembalikan instance yang sama — efisien untuk resource-heavy objects.

🔀 Kasus 3 – Mengembalikan Subclass (Shape)

Factory constructor bisa mengembalikan subclass berdasarkan parameter input.

abstract class Shape {
    void draw();

    // Factory constructor
    factory Shape(String type) {
        switch (type) {
            case 'circle':
                return Circle();
            case 'square':
                return Square();
            case 'triangle':
                return Triangle();
            case 'rectangle':
                return Rectangle();
            default:
                throw ArgumentError('Unknown shape: $type');
        }
    }
}

class Circle implements Shape {
    @override
    void draw() => print('Drawing Circle');
}

class Square implements Shape {
    @override
    void draw() => print('Drawing Square');
}

class Triangle implements Shape {
    @override
    void draw() => print('Drawing Triangle');
}

class Rectangle implements Shape {
    @override
    void draw() => print('Drawing Rectangle');
}

void main() {
    var circle = Shape('circle');
    var square = Shape('square');
    var triangle = Shape('triangle');
    var rectangle = Shape('rectangle');

    circle.draw();   // Drawing Circle
    square.draw();   // Drawing Square
    triangle.draw(); // Drawing Triangle
    rectangle.draw(); // Drawing Rectangle

    // var unknown = Shape('pentagon'); // throws ArgumentError
}
🔀 Factory constructor bertindak sebagai factory method yang memilih subclass yang tepat berdasarkan input, menjaga kode klien tetap sederhana.

⚙️ Praktik 1 – Factory untuk Configuration

Membuat factory untuk objek Config dengan berbagai source (file, environment, default).

class Config {
  final String apiUrl;
  final int timeout;
  final bool debug;

  // Private constructor
  Config._({
    required this.apiUrl,
    required this.timeout,
    required this.debug,
  });

  // Factory dari file JSON (simulasi)
  factory Config.fromJson(Map json) {
    return Config._(
      apiUrl: json['apiUrl'] ?? 'https://default.api',
      timeout: json['timeout'] ?? 30,
      debug: json['debug'] ?? false,
    );
  }

  // Factory dari environment (simulasi)
  factory Config.fromEnv() {
    // Dalam realita, membaca dari Platform.environment
    return Config._(
      apiUrl: 'https://env.api',
      timeout: 60,
      debug: true,
    );
  }

  // Factory default
  factory Config.defaultConfig() {
    return Config._(
      apiUrl: 'https://default.api',
      timeout: 30,
      debug: false,
    );
  }

  @override
  String toString() {
    return 'Config(apiUrl: $apiUrl, timeout: $timeout, debug: $debug)';
  }
}

void main() {
  var config1 = Config.fromJson({
    'apiUrl': 'https://prod.api',
    'timeout': 45,
  });

  var config2 = Config.fromEnv();
  var config3 = Config.defaultConfig();

  print(config1);
  print(config2);
  print(config3);
}
💡 Factory constructor memungkinkan pembuatan objek dari berbagai sumber dengan cara yang konsisten dan terstruktur.

🏷️ Factory dengan Named Constructors

Factory constructor bisa memiliki nama (seperti generative constructor). Ini memudahkan pembuatan objek dengan tujuan yang berbeda secara semantik, misalnya Response.success() dan Response.error().

abstract class Response {
  final T? data;
  final String? error;

  Response({this.data, this.error});

  // Factory dengan named constructor 'success'
  factory Response.success(T data) = SuccessResponse;

  // Factory dengan named constructor 'error'
  factory Response.error(String error) = ErrorResponse;
}

class SuccessResponse extends Response {
  SuccessResponse(T data) : super(data: data);
}

class ErrorResponse extends Response {
  ErrorResponse(String error) : super(error: error);
}

void main() {
  // Penggunaan named factory
  var success = Response.success('Data berhasil dimuat');
  var error = Response.error('Gagal memuat data');

  print('Success: data=${success.data}, error=${success.error}');
  print('Error: data=${error.data}, error=${error.error}');
}
📌 Named constructors pada factory membuat kode lebih ekspresif dan mudah dibaca, mirip dengan DateTime.now() atau Uri.parse().

🧬 Factory dengan Generic

Factory constructor dapat digunakan pada tipe generik (<T>) untuk memastikan type safety. Dengan redirecting factory, kita dapat mengembalikan subclass yang sesuai dengan tipe data.

// M BAGAS KHOIRULLAH (251410016) SI2A 
abstract class Response {
  final T? data;
  final String? error;

  Response({this.data, this.error});

  // Redirecting factory ke subclass dengan generic T
  factory Response.success(T data) = SuccessResponse;
  factory Response.error(String error) = ErrorResponse;
}

class SuccessResponse extends Response {
  SuccessResponse(T data) : super(data: data);
}

class ErrorResponse extends Response {
  ErrorResponse(String error) : super(error: error);
}

void main() {
  // Tipe data ditentukan saat pemanggilan
  var intResponse = Response.success(100);
  var stringResponse = Response.success('Hello');
  var errorResponse = Response.error('Failed');

  print(
    'intResponse data: ${intResponse.data} '
    '(type: ${intResponse.data.runtimeType})',
  );

  print(
    'stringResponse data: ${stringResponse.data} '
    '(type: ${stringResponse.data.runtimeType})',
  );

  print('errorResponse error: ${errorResponse.error}');
}
🧩 Generic pada factory memungkinkan pembuatan objek yang type-safe tanpa kehilangan fleksibilitas, sekaligus mendukung redirecting ke subclass.

💳 Praktik 2 – Factory untuk Payment Methods

Membuat factory yang memilih subclass payment berdasarkan tipe pembayaran, dengan validasi dan cache sederhana.

abstract class Payment {
  void pay(double amount);
  String get method;
}

class CreditCardPayment implements Payment {
  final String cardNumber;

  CreditCardPayment(this.cardNumber) {
    if (cardNumber.length < 16) {
      throw ArgumentError('Invalid card number');
    }
  }

  @override
  void pay(double amount) =>
      print('Paid \$$amount using Credit Card ($cardNumber)');

  @override
  String get method => 'CreditCard';
}

class PayPalPayment implements Payment {
  final String email;

  PayPalPayment(this.email) {
    if (!email.contains('@')) {
      throw ArgumentError('Invalid email');
    }
  }

  @override
  void pay(double amount) =>
      print('Paid \$$amount using PayPal ($email)');

  @override
  String get method => 'PayPal';
}

class CryptoPayment implements Payment {
  final String walletAddress;

  CryptoPayment(this.walletAddress) {
    if (walletAddress.length < 20) {
      throw ArgumentError('Invalid wallet address');
    }
  }

  @override
  void pay(double amount) =>
      print('Paid \$$amount using Crypto ($walletAddress)');

  @override
  String get method => 'Crypto';
}

class PaymentFactory {
  static final Map _cache = {};

  static Payment create(String type, String identifier) {
    var key = '$type:$identifier';

    if (_cache.containsKey(key)) {
      print('Returning cached payment: $key');
      return _cache[key]!;
    }

    Payment payment;

    switch (type.toLowerCase()) {
      case 'creditcard':
        payment = CreditCardPayment(identifier);
        break;

      case 'paypal':
        payment = PayPalPayment(identifier);
        break;

      case 'crypto':
        payment = CryptoPayment(identifier);
        break;

      default:
        throw ArgumentError('Unknown payment type: $type');
    }

    _cache[key] = payment;
    return payment;
  }
}

void main() {
  var cc1 = PaymentFactory.create(
      'creditcard', '1234567812345678');

  var cc2 = PaymentFactory.create(
      'creditcard', '1234567812345678'); // cached

  var pp = PaymentFactory.create(
      'paypal', 'user@example.com');

  var crypto = PaymentFactory.create(
      'crypto', '0xabcdef12345678901234');

  cc1.pay(100.0);
  pp.pay(50.0);
  crypto.pay(200.0);

  print('cc1 dan cc2 objek yang sama? ${identical(cc1, cc2)}');
}
💳 Factory menangani validasi, cache, dan pemilihan subclass — semua dalam satu titik pembuatan objek.

🎨 Factory untuk Immutable Objects dengan Cache (Color)

Menggunakan factory constructor untuk meng-cache objek immutable (Color) sehingga instance yang sama digunakan untuk nilai yang sama.

// M BAGAS KHOIRULLAH (251410016) SI2A 
class Color {
  final int red;
  final int green;
  final int blue;

  // Cache immutable objects
  static final Map _cache = {};

  // Private constructor
  Color._(this.red, this.green, this.blue);

  // Factory dengan cache
  factory Color.rgb(int red, int green, int blue) {
    // Validasi
    if (red < 0 ||
        red > 255 ||
        green < 0 ||
        green > 255 ||
        blue < 0 ||
        blue > 255) {
      throw ArgumentError('RGB values must be 0-255');
    }

    var key = '$red,$green,$blue';

    if (!_cache.containsKey(key)) {
      _cache[key] = Color._(red, green, blue);
    }

    return _cache[key]!;
  }

  // Named factories untuk warna umum
  factory Color.red() => Color.rgb(255, 0, 0);
  factory Color.green() => Color.rgb(0, 255, 0);
  factory Color.blue() => Color.rgb(0, 0, 255);
  factory Color.white() => Color.rgb(255, 255, 255);
  factory Color.black() => Color.rgb(0, 0, 0);

  @override
  String toString() => 'Color($red, $green, $blue)';
}

void main() {
  var red1 = Color.red();
  var red2 = Color.rgb(255, 0, 0);

  print(identical(red1, red2)); // true - instance sama
  print(red1);
  print(red2);
}
🎯 Dengan cache, warna yang sama tidak dibuat berulang kali — menghemat memori untuk objek yang immutable dan sering digunakan.

📄 Studi Kasus – Document Factory

Factory untuk membuat berbagai jenis dokumen (Word, PDF, Excel) dengan cache dan validasi ekstensi file.

// M BAGAS KHOIRULLAH (251410016) SI2A 
abstract class Document {
  void open();
  void save();
  String get extension;
}

class WordDocument implements Document {
  final String fileName;

  WordDocument(this.fileName) {
    if (!fileName.endsWith('.docx')) {
      throw ArgumentError('Must be .docx');
    }
  }

  @override
  void open() => print('Opening Word document: $fileName');

  @override
  void save() => print('Saving Word document: $fileName');

  @override
  String get extension => '.docx';
}

class PdfDocument implements Document {
  final String fileName;

  PdfDocument(this.fileName) {
    if (!fileName.endsWith('.pdf')) {
      throw ArgumentError('Must be .pdf');
    }
  }

  @override
  void open() => print('Opening PDF document: $fileName');

  @override
  void save() => print('Saving PDF document: $fileName');

  @override
  String get extension => '.pdf';
}

class ExcelDocument implements Document {
  final String fileName;

  ExcelDocument(this.fileName) {
    if (!fileName.endsWith('.xlsx')) {
      throw ArgumentError('Must be .xlsx');
    }
  }

  @override
  void open() => print('Opening Excel document: $fileName');

  @override
  void save() => print('Saving Excel document: $fileName');

  @override
  String get extension => '.xlsx';
}

class DocumentFactory {
  static final Map _cache = {};

  static Document create(String fileName) {
    var key = fileName.toLowerCase();

    if (_cache.containsKey(key)) {
      print('Returning cached document: $key');
      return _cache[key]!;
    }

    Document doc;

    if (fileName.endsWith('.docx')) {
      doc = WordDocument(fileName);
    } else if (fileName.endsWith('.pdf')) {
      doc = PdfDocument(fileName);
    } else if (fileName.endsWith('.xlsx')) {
      doc = ExcelDocument(fileName);
    } else {
      throw ArgumentError('Unsupported file type: $fileName');
    }

    _cache[key] = doc;
    return doc;
  }
}

void main() {
  var word1 = DocumentFactory.create('report.docx');
  var word2 = DocumentFactory.create('report.docx'); // cached
  var pdf = DocumentFactory.create('presentation.pdf');
  var excel = DocumentFactory.create('data.xlsx');

  word1.open();
  pdf.open();
  excel.open();

  print(identical(word1, word2)); // true

  // Contoh error:
  /*
  try {
    DocumentFactory.create('file.txt');
  } catch (e) {
    print(e);
  }
  */
}
📄 Factory ini menangani validasi ekstensi, cache, dan pembuatan subclass yang sesuai — sangat berguna dalam aplikasi pengolah dokumen.

⚖️ Factory Constructor vs Static Method

Perbedaan utama: Factory constructor dapat digunakan di new expression dan mendukung polymorphism.

class Example {
    Example._() {}

    // Factory constructor
    factory Example.create() => Example._();

    // Static method
    static Example createStatic() => Example._();
}

void main() {
    // Penggunaan sama secara sintaks
    var e1 = Example.create();       // Factory constructor
    var e2 = Example.createStatic(); // Static method

    print('e1: $e1');
    print('e2: $e2');
    print('Apakah e1 dan e2 berbeda? ${e1 != e2}');
}
AspekFactory ConstructorStatic Method
Digunakan dengan new✅ Ya❌ Tidak
Dapat mengembalikan subclass✅ Ya✅ Ya
Mendukung inheritance✅ Ya❌ Tidak
Membuat instance baru⚡ Opsional⚡ Opsional
💡 Gunakan factory constructor jika ingin objek dibuat dengan syntax seperti constructor dan mendukung polimorfisme. Gunakan static method untuk utility sederhana.

📝 Ringkasan & Best Practices

Ringkasan

  • Factory constructor: factory Class.name()
  • Bisa mengembalikan: instance yang ada, subclass, atau instance baru.
  • Use cases: Singleton, caching, object pooling, subclass selection, immutable objects.

Best Practices

  • Gunakan factory untuk:
    • Object pooling / caching
    • Singleton pattern
    • Complex object creation
    • Subclass selection
    • Immutable object caching
  • Gunakan generative constructor untuk:
    • Simple object creation
    • Initialization lists
    • Required parameters
  • Validation: Lakukan validasi di factory sebelum membuat objek.
⚡ Ingat: Factory constructor tidak memiliki akses ke this sebelum objek dibuat — ia hanya bertugas mengembalikan instance.

// M BAGAS KHOIRULLAH (251410016) SI2A 
abstract class Animal {
  String get name;
  String speak();
}

class Dog implements Animal {
  final String _name;

  Dog(this._name) {
    if (_name.isEmpty) throw ArgumentError('Name cannot be empty');
  }

  @override
  String get name => _name;

  @override
  String speak() => 'Woof!';
}

class Cat implements Animal {
  final String _name;

  Cat(this._name) {
    if (_name.isEmpty) throw ArgumentError('Name cannot be empty');
  }

  @override
  String get name => _name;

  @override
  String speak() => 'Meow!';
}

class Cow implements Animal {
  final String _name;

  Cow(this._name) {
    if (_name.isEmpty) throw ArgumentError('Name cannot be empty');
  }

  @override
  String get name => _name;

  @override
  String speak() => 'Moo!';
}

class AnimalFactory {
  static final Map _cache = {};

  static Animal create(String type, String name) {
    if (name.isEmpty) {
      throw ArgumentError('Animal name cannot be empty');
    }

    var key = '$type:$name';

    if (_cache.containsKey(key)) {
      return _cache[key]!;
    }

    Animal animal;

    switch (type.toLowerCase()) {
      case 'dog':
        animal = Dog(name);
        break;
      case 'cat':
        animal = Cat(name);
        break;
      case 'cow':
        animal = Cow(name);
        break;
      default:
        throw ArgumentError('Unknown animal type: $type');
    }

    _cache[key] = animal;
    return animal;
  }
}

void main() {
  var dog1 = AnimalFactory.create('dog', 'Buddy');
  var dog2 = AnimalFactory.create('dog', 'Buddy');
  var dog3 = AnimalFactory.create('dog', 'Rex');

  print(identical(dog1, dog2)); // true
  print(identical(dog1, dog3)); // false

  print('${dog1.name} says ${dog1.speak()}');
  print('${dog3.name} says ${dog3.speak()}');

  var cat = AnimalFactory.create('cat', 'Whiskers');
  print('${cat.name} says ${cat.speak()}');

  var cow = AnimalFactory.create('cow', 'Bessie');
  print('${cow.name} says ${cow.speak()}');

  try {
    AnimalFactory.create('dog', '');
  } catch (e) {
    print('Validation works: $e');
  }

  try {
    AnimalFactory.create('bird', 'Tweety');
  } catch (e) {
    print('Unknown type handled: $e');
  }
}
